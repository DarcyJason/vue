Vue

v-model

v-model 用于给表单元素进行双向绑定, 可以快速获取表单元素的值

<div id="app">
  账户: <input type="text" v-model="username"> <br><br>
  密码: <input type="password" v-model="password">
  <button>登录</button>
  <button>重置</button>
</div>

const app = new Vue({
  el: '#app',
  data: {
    username: '',
    password: '',
  },
  methods: {
    login () {
      console.log(this.username, this.password)
    },
    reset () {
      this.username = ''
      this.password = ''
    }
  }
})
